Given $20\% \Ket{-}_x$, $75\% \Ket{+}_y$, $5\% \Ket{+}_z$:

\begin{align*}
    \rho &= \sum_i p_i \Ket{\psi_i}\Bra{\psi_i} \\
         &= 0.2 
    \biggl( \Ket{-}_x 
    {}_x\Bra{-} \biggr)
 + 0.75 
    \biggl( \Ket{+}_y 
    {}_y\Bra{+} \biggr)
 + 0.05 
    \biggl( \Ket{+}_z 
    {}_z\Bra{+} \biggr)
 \\
         &= 0.2 \mathbf{-S_x} + 0.75 \mathbf{S_y} + 0.05 \mathbf{S_z} \\
         &= \frac{\hbar}{2} \begin{bmatrix}
    0.005 & -0.2 + 0.75i \\
    0.2 - 0.75i & -0.05
\end{bmatrix} \\ \\
    \Braket{\mathbf{S_y}\rho}_{\Ket{\psi}} &= \frac{\hbar^2}{4}Tr\biggl\{
\begin{bmatrix}
    0 & -i \\
    i & 0
\end{bmatrix}
\begin{bmatrix}
    0.005 & -0.2 + 0.75i \\
    0.2 - 0.75i & -0.05
\end{bmatrix}
    \biggl\} \\
    &= \frac{\hbar^2}{4} (0.75 + 0.2i + -0.75 -0.2i) \\
    &= 0
\end{align*}
